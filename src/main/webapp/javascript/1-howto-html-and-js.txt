/*
JavaScript
- JavaScript can be placed in the <body> and the <head> sections of an HTML page.
- in HTML, JavaScript code must be inserted between <script> and </script> tags.
- scripts can also be invoked from external .js files: */
  <script src="myScript.js"></script>
/*
HTML and JavaScript
- external scripts cannot contain <script> tags.
- older examples may use a type  attribute: <script type="text/javascript">.
- this type attribute is not required. JavaScript is the default scripting language in HTML.
- using id="demo" in a HTML element JS can use innerHTML to display data at a specific point
- HTML allows events with JavaScript code to be added to HTML elements.
- HTML: */
<p id="demo">Temperature: -</p>
<input id="inputTemperature" type="number" min="1" max="100">
<button id="btnFahrenheid" onclick="toFahrenheid()">toFahrenheid</button>
<button id="btnCelcius" onclick="toCelsius()">toCelsius</button>
<p id="error"></p>
/*
- JS: in myScript.js*/
function toFahrenheid() {
 document.getElementById("error").innerHTML = ""
 document.getElementById("demo").innerHTML = "Temperature: -"
 if (document.getElementById("inputTemperature").validity.rangeOverflow) {
     document.getElementById("error").innerHTML = "Input invalid";
  } else {
  var input = document.getElementById("inputTemperature").value;
  var temperature = ((1.8 * input) + 32);
  document.getElementById("demo").innerHTML = "Temperature: " + temperature.toFixed(1) + " fahrenheid";
  }
}
function toCelsius() {
 document.getElementById("error").innerHTML = ""
 document.getElementById("demo").innerHTML = "Temperature: -"
 if (document.getElementById("inputTemperature").validity.rangeOverflow) {
     document.getElementById("error").innerHTML = "Input invalid";
  } else {
     var input = document.getElementById("inputTemperature").value;
     var temperature = ((input - 32) / 1.8 );
     document.getElementById("demo").innerHTML = "Temperature: " + temperature.toFixed(1) + " celcius";
  }
}
